255  protected static Xid[] recover(Connection c, int flag) throws XAException {
256  /*
257  The XA RECOVER statement returns information for those XA transactions on the MySQL server that are in the PREPARED state. (See Section 13.4.7.2, �XA Transaction States�.) The output includes a row for each such XA transaction on the server, regardless of which client started it.
258 
259  XA RECOVER output rows look like this (for an example xid value consisting of the parts 'abc', 'def', and 7):
260 
261  mysql> XA RECOVER;
262  +----------+--------------+--------------+--------+
263  | formatID | gtrid_length | bqual_length | data |
264  +----------+--------------+--------------+--------+
265  | 7 | 3 | 3 | abcdef |
266  +----------+--------------+--------------+--------+
267 
268  The output columns have the following meanings:
269 
270  formatID is the formatID part of the transaction xid
271  gtrid_length is the length in bytes of the gtrid part of the xid
272  bqual_length is the length in bytes of the bqual part of the xid
273  data is the concatenation of the gtrid and bqual parts of the xid
274  */
275 
276  boolean startRscan = ((flag & TMSTARTRSCAN) > 0);
277  boolean endRscan = ((flag & TMENDRSCAN) > 0);
278 
279  if (!startRscan && !endRscan && flag != TMNOFLAGS) {
280  throw new MysqlXAException(XAException.XAER_INVAL,
281  "Invalid flag, must use TMNOFLAGS, or any combination of TMSTARTRSCAN and TMENDRSCAN",
282  null);
283  }
284 
285  //
286  // We return all recovered XIDs at once, so if not
287  // TMSTARTRSCAN, return no new XIDs
288  //
289  // We don't attempt to maintain state to check for TMNOFLAGS
290  // "outside" of a scan
291  //
292 
293  if (!startRscan) {
294  return new Xid[0];
295  }
296 
297  ResultSet rs = null;
298  Statement stmt = null;
299 
300  List<MysqlXid> recoveredXidList = new ArrayList<MysqlXid>();
301 
302  try {
303  // TODO: Cache this for lifetime of XAConnection
304  stmt = c.createStatement();
305 
306  rs = stmt.executeQuery("XA RECOVER");
307 
308  while (rs.next()) {
309  final int formatId = rs.getInt(1);
310  int gtridLength = rs.getInt(2);
311  int bqualLength = rs.getInt(3);
312  byte[] gtridAndBqual = rs.getBytes(4);
313 
314  final byte[] gtrid = new byte[gtridLength];
315  final byte[] bqual = new byte[bqualLength];
316 
317  if (gtridAndBqual.length != (gtridLength + bqualLength)) {
318  throw new MysqlXAException(XAException.XA_RBPROTO,
319  "Error while recovering XIDs from RM. GTRID and BQUAL are wrong sizes",
320  null);
321  }
322 
323  System.arraycopy(gtridAndBqual, 0, gtrid, 0,
324  gtridLength);
325  System.arraycopy(gtridAndBqual, gtridLength, bqual, 0,
326  bqualLength);
327 
328  recoveredXidList.add(new MysqlXid(gtrid, bqual,
329  formatId));
330  }
331  } catch (SQLException sqlEx) {
332  throw mapXAExceptionFromSQLException(sqlEx);
333  } finally {
334  if (rs != null) {
335  try {
336  rs.close();
337  } catch (SQLException sqlEx) {
338  throw mapXAExceptionFromSQLException(sqlEx);
339  }
340  }
341 
342  if (stmt != null) {
343  try {
344  stmt.close();
345  } catch (SQLException sqlEx) {
346  throw mapXAExceptionFromSQLException(sqlEx);
347  }
348  }
349  }
350 
351  int numXids = recoveredXidList.size();
352 
353  Xid[] asXids = new Xid[numXids];
354  Object[] asObjects = recoveredXidList.toArray();
355 
356  for (int i = 0; i < numXids; i++) {
357  asXids[i] = (Xid) asObjects[i];
358  }
359 
360  return asXids;
361  }

608  private static String xidToString(Xid xid) {
609  byte[] gtrid = xid.getGlobalTransactionId();
610 
611  byte[] btrid = xid.getBranchQualifier();
612 
613  int lengthAsString = 6; // for (0x and ,) * 2
614 
615  if (gtrid != null) {
616  lengthAsString += (2 * gtrid.length);
617  }
618 
619  if (btrid != null) {
620  lengthAsString += (2 * btrid.length);
621  }
622 
623  String formatIdInHex = Integer.toHexString(xid.getFormatId());
624 
625  lengthAsString += formatIdInHex.length();
626  lengthAsString += 3; // for the '.' after formatId
627 
628  StringBuffer asString = new StringBuffer(lengthAsString);
629 
630  asString.append("0x");
631 
632  if (gtrid != null) {
633  for (int i = 0; i < gtrid.length; i++) {
634  String asHex = Integer.toHexString(gtrid[i] & 0xff);
635 
636  if (asHex.length() == 1) {
637  asString.append("0");
638  }
639 
640  asString.append(asHex);
641  }
642  }
643 
644  asString.append(",");
645 
646  if (btrid != null) {
647  asString.append("0x");
648 
649  for (int i = 0; i < btrid.length; i++) {
650  String asHex = Integer.toHexString(btrid[i] & 0xff);
651 
652  if (asHex.length() == 1) {
653  asString.append("0");
654  }
655 
656  asString.append(asHex);
657  }
658  }
659 
660  asString.append(",0x");
661  asString.append(formatIdInHex);
662 
663  return asString.toString();
664  }
