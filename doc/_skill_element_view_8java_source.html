249  public static int calculateInSampleSize(BitmapFactory.Options options,
250  int reqWidth, int reqHeight) {
251  // Raw height and width of image
252  final int height = options.outHeight;
253  final int width = options.outWidth;
254  int inSampleSize = 1;
255 
256  if (height > reqHeight || width > reqWidth) {
257 
258  final int halfHeight = height / 2;
259  final int halfWidth = width / 2;
260 
261  // Calculate the largest inSampleSize value that is a power of 2 and
262  // keeps both
263  // height and width larger than the requested height and width.
264  while ((halfHeight / inSampleSize) > reqHeight
265  && (halfWidth / inSampleSize) > reqWidth) {
266  inSampleSize *= 2;
267  }
268  }
269
