195  public void testDeadlockDetection() throws Exception {

199  this.rs.next();

204  this.stmt.executeUpdate("INSERT INTO t1 VALUES(0, 0)");
205  this.conn.setAutoCommit(false);
206 
207  Properties props = new Properties();
208  props.setProperty("includeInnodbStatusInDeadlockExceptions", "true");
209  props.setProperty("includeThreadDumpInDeadlockExceptions", "true");
210 
211  Connection deadlockConn = getConnectionWithProps(props);
212  deadlockConn.setAutoCommit(false);
213 
214  try {
215  this.conn.createStatement().executeQuery(
216  "SELECT * FROM t1 WHERE id=0 FOR UPDATE");
217 
218  // The following query should hang because con1 is locking the
219  // page
220  deadlockConn.createStatement().executeUpdate(
221  "UPDATE t1 SET x=2 WHERE id=0");
222  } finally {
223  if (versionMeetsMinimum(5, 5)) {
224  this.conn.commit();
225  deadlockConn.commit();
226  }
227  }
228 
229  Thread.sleep(timeoutSecs * 2 * 1000);
230  } catch (SQLException sqlEx) {
231  System.out
232  .println("Caught SQLException due to deadlock/lock timeout");
233  System.out.println("SQLState: " + sqlEx.getSQLState());
234  System.out.println("Vendor error: " + sqlEx.getErrorCode());
235  System.out.println("Message: " + sqlEx.getMessage());
236

239  //

241  assertTrue(sqlEx.getErrorCode() == 1205);
242  // Make sure INNODB Status is getting dumped into error message
243 
244  if (sqlEx.getMessage().indexOf("PROCESS privilege") != -1) {
245  fail("This test requires user with process privilege");
246  }
247 
248  assertTrue(

250  sqlEx.getMessage().indexOf("INNODB MONITOR") != -1);
251

253  "Can't find thread dump in:\n\n" + sqlEx.getMessage(),
254  sqlEx.getMessage().indexOf("testsuite.simple.ConnectionTest.testDeadlockDetection") != -1);
