658  public Properties parseURL(String url, Properties defaults)
659  throws java.sql.SQLException {
660  Properties urlProps = (defaults != null) ? new Properties(defaults)
661  : new Properties();
662 
663  if (url == null) {
664  return null;
665  }
666 
667  if (!StringUtils.startsWithIgnoreCase(url, URL_PREFIX)
668  && !StringUtils.startsWithIgnoreCase(url, MXJ_URL_PREFIX)
669  && !StringUtils.startsWithIgnoreCase(url,
670  LOADBALANCE_URL_PREFIX)
671  && !StringUtils.startsWithIgnoreCase(url,
672  REPLICATION_URL_PREFIX)) { //$NON-NLS-1$
673 
674  return null;
675  }
676 
677  int beginningOfSlashes = url.indexOf("//");
678 
679  if (StringUtils.startsWithIgnoreCase(url, MXJ_URL_PREFIX)) {
680 
681  urlProps
682  .setProperty("socketFactory",
683  "com.mysql.management.driverlaunched.ServerLauncherSocketFactory");
684  }
685 
686  /*
687  * Parse parameters after the ? in the URL and remove them from the
688  * original URL.
689  */
690  int index = url.indexOf("?"); //$NON-NLS-1$
691 
692  if (index != -1) {
693  String paramString = url.substring(index + 1, url.length());
694  url = url.substring(0, index);
695 
696  StringTokenizer queryParams = new StringTokenizer(paramString, "&"); //$NON-NLS-1$
697 
698  while (queryParams.hasMoreTokens()) {
699  String parameterValuePair = queryParams.nextToken();
700 
701  int indexOfEquals = StringUtils.indexOfIgnoreCase(0,
702  parameterValuePair, "=");
703 
704  String parameter = null;
705  String value = null;
706 
707  if (indexOfEquals != -1) {
708  parameter = parameterValuePair.substring(0, indexOfEquals);
709 
710  if (indexOfEquals + 1 < parameterValuePair.length()) {
711  value = parameterValuePair.substring(indexOfEquals + 1);
712  }
713  }
714 
715  if ((value != null && value.length() > 0)
716  && (parameter != null && parameter.length() > 0)) {
717  try {
718  urlProps.put(parameter, URLDecoder.decode(value,
719  "UTF-8"));
720  } catch (UnsupportedEncodingException badEncoding) {
721  // punt
722  urlProps.put(parameter, URLDecoder.decode(value));
723  } catch (NoSuchMethodError nsme) {
724  // punt again
725  urlProps.put(parameter, URLDecoder.decode(value));
726  }
727  }
728  }
729  }
730 
731  url = url.substring(beginningOfSlashes + 2);
732 
733  String hostStuff = null;
734 
735  int slashIndex = StringUtils.indexOfIgnoreCaseRespectMarker(0, url, "/", ALLOWED_QUOTES, ALLOWED_QUOTES, true); //$NON-NLS-1$
736 
737  if (slashIndex != -1) {
738  hostStuff = url.substring(0, slashIndex);
739 
740  if ((slashIndex + 1) < url.length()) {
741  urlProps.put(DBNAME_PROPERTY_KEY, //$NON-NLS-1$
742  url.substring((slashIndex + 1), url.length()));
743  }
744  } else {
745  hostStuff = url;
746  }
747 
748  int numHosts = 0;
749 
750  if ((hostStuff != null) && (hostStuff.trim().length() > 0)) {
751  List<String> hosts = StringUtils.split(hostStuff, ",", ALLOWED_QUOTES, ALLOWED_QUOTES, false);
752 
753 
754  for (String hostAndPort : hosts) {
755  numHosts++;
756 
757  String[] hostPortPair = parseHostPortPair(hostAndPort);
758 
759  if (hostPortPair[HOST_NAME_INDEX] != null && hostPortPair[HOST_NAME_INDEX].trim().length() > 0) {
760  urlProps.setProperty(HOST_PROPERTY_KEY + "." + numHosts, hostPortPair[HOST_NAME_INDEX]);
761  } else {
762  urlProps.setProperty(HOST_PROPERTY_KEY + "." + numHosts, "localhost");
763  }
764 
765  if (hostPortPair[PORT_NUMBER_INDEX] != null) {
766  urlProps.setProperty(PORT_PROPERTY_KEY + "." + numHosts, hostPortPair[PORT_NUMBER_INDEX]);
767  } else {
768  urlProps.setProperty(PORT_PROPERTY_KEY + "." + numHosts, "3306");
769  }
770  }
771  } else {
772  numHosts = 1;
773  urlProps.setProperty(HOST_PROPERTY_KEY + ".1", "localhost");
774  urlProps.setProperty(PORT_PROPERTY_KEY + ".1", "3306");
775  }
776 
777  urlProps.setProperty(NUM_HOSTS_PROPERTY_KEY, String.valueOf(numHosts));
778  urlProps.setProperty(HOST_PROPERTY_KEY, urlProps.getProperty(HOST_PROPERTY_KEY + ".1"));
779  urlProps.setProperty(PORT_PROPERTY_KEY, urlProps.getProperty(PORT_PROPERTY_KEY + ".1"));
780 
781  String propertiesTransformClassName = urlProps
782  .getProperty(PROPERTIES_TRANSFORM_KEY);
783 
784  if (propertiesTransformClassName != null) {
785  try {
786  ConnectionPropertiesTransform propTransformer = (ConnectionPropertiesTransform) Class
787  .forName(propertiesTransformClassName).newInstance();
788 
789  urlProps = propTransformer.transformProperties(urlProps);
790  } catch (InstantiationException e) {
791  throw SQLError.createSQLException(
792  "Unable to create properties transform instance '"
793  + propertiesTransformClassName
794  + "' due to underlying exception: "
795  + e.toString(),
796  SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE, null);
797  } catch (IllegalAccessException e) {
798  throw SQLError.createSQLException(
799  "Unable to create properties transform instance '"
800  + propertiesTransformClassName
801  + "' due to underlying exception: "
802  + e.toString(),
803  SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE, null);
804  } catch (ClassNotFoundException e) {
805  throw SQLError.createSQLException(
806  "Unable to create properties transform instance '"
807  + propertiesTransformClassName
808  + "' due to underlying exception: "
809  + e.toString(),
810  SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE, null);
811  }
812  }
813 
814  if (Util.isColdFusion() &&
815  urlProps.getProperty("autoConfigureForColdFusion", "true").equalsIgnoreCase("true")) {
816  String configs = urlProps.getProperty(USE_CONFIG_PROPERTY_KEY);
817 
818  StringBuffer newConfigs = new StringBuffer();
819 
820  if (configs != null) {
821  newConfigs.append(configs);
822  newConfigs.append(",");
823  }
824 
825  newConfigs.append("coldFusion");
826 
827  urlProps.setProperty(USE_CONFIG_PROPERTY_KEY, newConfigs.toString());
828  }
829

831  // the URL or passed-in properties
832 
833  String configNames = null;
834 
835  if (defaults != null) {
836  configNames = defaults.getProperty(USE_CONFIG_PROPERTY_KEY);
837  }
838 
839  if (configNames == null) {
840  configNames = urlProps.getProperty(USE_CONFIG_PROPERTY_KEY);
841  }
842 
843  if (configNames != null) {
844  List<String> splitNames = StringUtils.split(configNames, ",", true);
845 
846  Properties configProps = new Properties();
847 
848  Iterator<String> namesIter = splitNames.iterator();
849 
850  while (namesIter.hasNext()) {
851  String configName = namesIter.next();
852 
853  try {
854  InputStream configAsStream = getClass()
855  .getResourceAsStream(
856  "configs/" + configName + ".properties");
857 
858  if (configAsStream == null) {
859  throw SQLError

863  SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE, null);
864  }
865  configProps.load(configAsStream);
866  } catch (IOException ioEx) {
867  SQLException sqlEx = SQLError.createSQLException(

871  + ioEx,
872  SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE, null);
873  sqlEx.initCause(ioEx);
874 
875  throw sqlEx;
876  }
877  }
878 
879  Iterator<Object> propsIter = urlProps.keySet().iterator();
880 
881  while (propsIter.hasNext()) {
882  String key = propsIter.next().toString();
883  String property = urlProps.getProperty(key);
884  configProps.setProperty(key, property);
885  }
886 
887  urlProps = configProps;
888  }
