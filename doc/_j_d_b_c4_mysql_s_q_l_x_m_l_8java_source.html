537  private Reader binaryInputStreamStreamToReader(ByteArrayOutputStream out) {
538 
539  try {
540  // There's got to be an easier way to do this, but
541  // I don't feel like coding up Appendix F of the XML Spec
542  // myself, when there's a reusable way to do it, and we
543  // can warn folks away from BINARY xml streams that have
544  // to be parsed to determine the character encoding :P
545 
546  String encoding = "UTF-8";
547 
548  try {
549  ByteArrayInputStream bIn = new ByteArrayInputStream(out
550  .toByteArray());
551  XMLStreamReader reader = this.inputFactory
552  .createXMLStreamReader(bIn);
553 
554  int eventType = 0;
555 
556  while ((eventType = reader.next()) != XMLStreamReader.END_DOCUMENT) {
557  if (eventType == XMLStreamReader.START_DOCUMENT) {
558  String possibleEncoding = reader.getEncoding();
559 
560  if (possibleEncoding != null) {
561  encoding = possibleEncoding;
562  }
563 
564  break;
565  }
566  }
567  } catch (Throwable t) {
568  // ignore, dealt with later when the string can't be parsed
569  // into valid XML
570  }
571 
572  return new StringReader(new String(out.toByteArray(), encoding));
573  } catch (UnsupportedEncodingException badEnc) {
574  throw new RuntimeException(badEnc);
575  }
576  }

792  private void escapeCharsForXml(char c, boolean isAttributeData) {
793  switch (c) {
794  case '<':
795  this.buf.append("&lt;");
796  break;
797 
798  case '>':
799  this.buf.append("&gt;");
800  break;
801 
802  case '&':
803  this.buf.append("&amp;");
804  break;
805 
806  case '"':
807 
808  if (!isAttributeData) {
809  this.buf.append("\"");
810  }
811  else {
812  this.buf.append("&quot;");
813  }
814 
815  break;
816 
817  case '\r':
818  this.buf.append("&#xD;");
819  break;
820 
821 
822  default:
823 
824  if (((c >= 0x01 && c <= 0x1F && c != 0x09 && c != 0x0A)
825  || (c >= 0x7F && c <= 0x9F) || c == 0x2028)
826  || isAttributeData && (c == 0x09 || c == 0x0A)) {
827  this.buf.append("&#x");
828  this.buf.append(Integer.toHexString(c).toUpperCase());
829  this.buf.append(";");
830  }
831  else {
832  this.buf.append(c);
833  }
834  }
835  }
