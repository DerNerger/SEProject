729  static String getClassNameForJavaType(int javaType, boolean isUnsigned, int mysqlTypeIfKnown,

733  case Types.BOOLEAN:
734  return "java.lang.Boolean";

761  case Types.BIGINT:
762

764  return "java.lang.Long";

768 
769  case Types.DECIMAL:
770  case Types.NUMERIC:
771  return "java.math.BigDecimal";
772 
773  case Types.REAL:
774  return "java.lang.Float";
775 
776  case Types.FLOAT:
777  case Types.DOUBLE:
778  return "java.lang.Double";
779 
780  case Types.CHAR:
781  case Types.VARCHAR:
782  case Types.LONGVARCHAR:
783  if (!isOpaqueBinary) {
784  return "java.lang.String";
785  }
786 
787  return "[B";
788 
789  case Types.BINARY:
790  case Types.VARBINARY:
791  case Types.LONGVARBINARY:
792 
793  if (mysqlTypeIfKnown == MysqlDefs.FIELD_TYPE_GEOMETRY) {
794  return "[B";
795  } else if (isBinaryOrBlob) {
796  return "[B";
797  } else {
798  return "java.lang.String";
799  }
800 
801  case Types.DATE:
802  return (treatYearAsDate || mysqlTypeIfKnown != MysqlDefs.FIELD_TYPE_YEAR) ? "java.sql.Date"
803  : "java.lang.Short";
804 
805  case Types.TIME:
806  return "java.sql.Time";
807 
808  case Types.TIMESTAMP:
809  return "java.sql.Timestamp";
810 
811  default:
