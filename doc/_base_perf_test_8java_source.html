167  protected void addResult(double value) {
168  this.numIterations++;
169  this.testValuesList.add(new Double(value));
170 
171  this.sumValue += value;
172  this.squareSumValue += (value * value);
173  this.meanValue = this.sumValue / this.numIterations;
174  this.variationValue = (this.squareSumValue / this.numIterations)
175  - (this.meanValue * this.meanValue);
176 
177  // Can only have confidence when more than one test
178  // has been completed
179  if (this.numIterations > 1) {
180  this.confidenceValue = this.intervalWidth
181  - ((2.0 * getConfidenceLookup() * Math
182  .sqrt(this.variationValue
183  / (this.numIterations - 1.0))) / this.meanValue);
