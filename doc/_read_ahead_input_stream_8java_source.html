58  private void fill(int readAtLeastTheseManyBytes) throws IOException {
59  checkClosed();

61  this.currentPosition = 0; /* no mark: throw away the buffer */

178  public synchronized int read(byte b[], int off, int len) throws IOException {
179  checkClosed(); // Check for closed stream
180  if ((off | len | (off + len) | (b.length - (off + len))) < 0) {
181  throw new IndexOutOfBoundsException();
182  } else if (len == 0) {
183  return 0;
184  }
185 
186  int totalBytesRead = 0;
187 
188  while (true) {
189  int bytesReadThisRound = readFromUnderlyingStreamIfNecessary(b, off
190  + totalBytesRead, len - totalBytesRead);
191 
192  // end-of-stream?
193  if (bytesReadThisRound <= 0) {
194  if (totalBytesRead == 0) {
195  totalBytesRead = bytesReadThisRound;
196  }
197 
198  break;
199  }
200 
201  totalBytesRead += bytesReadThisRound;
202 
203  // Read _at_least_ enough bytes
204  if (totalBytesRead >= len) {
205  break;
206  }
207 
208  // Nothing to read?
209  if (this.underlyingStream.available() <= 0) {
210  break;
211  }
212  }
213 
214  return totalBytesRead;
