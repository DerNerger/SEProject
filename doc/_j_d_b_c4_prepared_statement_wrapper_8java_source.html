503  public synchronized <T> T unwrap(java.lang.Class<T> iface)
504  throws java.sql.SQLException {
505  try {
506  if ("java.sql.Statement".equals(iface.getName())
507  || "java.sql.PreparedStatement".equals(iface.getName())
508  || "java.sql.Wrapper.class".equals(iface.getName())) {
509  return iface.cast(this);
510  }
511 
512  if (unwrappedInterfaces == null) {
513  unwrappedInterfaces = new HashMap();
514  }
515 
516  Object cachedUnwrapped = unwrappedInterfaces.get(iface);
517 
518  if (cachedUnwrapped == null) {
519  if (cachedUnwrapped == null) {
520  cachedUnwrapped = Proxy.newProxyInstance(
521  this.wrappedStmt.getClass().getClassLoader(),
522  new Class[] { iface },
523  new ConnectionErrorFiringInvocationHandler(this.wrappedStmt));
524  unwrappedInterfaces.put(iface, cachedUnwrapped);
525  }
526  unwrappedInterfaces.put(iface, cachedUnwrapped);
527  }
528 
529  return iface.cast(cachedUnwrapped);
530  } catch (ClassCastException cce) {
531  throw SQLError.createSQLException("Unable to unwrap to "
532  + iface.toString(), SQLError.SQL_STATE_ILLEGAL_ARGUMENT, this.exceptionInterceptor);
