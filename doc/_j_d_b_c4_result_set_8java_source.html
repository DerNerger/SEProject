171  private String getStringForNClob(int columnIndex) throws SQLException {
172  String asString = null;
173 
174  String forcedEncoding = "UTF-8";
175 
176  try {
177  byte[] asBytes = null;
178 
179  if (!this.isBinaryEncoded) {
180  asBytes = getBytes(columnIndex);
181  } else {
182  asBytes = getNativeBytes(columnIndex, true);
183  }
184 
185  if (asBytes != null) {
186  asString = new String(asBytes, forcedEncoding);
187  }
188  } catch (UnsupportedEncodingException uee) {
189  throw SQLError.createSQLException("Unsupported character encoding "
190  + forcedEncoding, SQLError.SQL_STATE_ILLEGAL_ARGUMENT, getExceptionInterceptor());
191  }
192 
193  return asString;
194  }
